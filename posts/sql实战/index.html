本文基于停留的风-SQL实战中的建表脚本和问题，使用MySQL语法实现。

一、建表语句

DROP TABLE IF EXISTS student;
CREATE TABLE student(
sid VARCHAR(20) PRIMARY KEY COMMENT '学号',
sname VARCHAR(20) COMMENT '姓名',
sage INT COMMENT '年龄',
ssex VARCHAR(1) COMMENT '性别'
)ENGINE=INNODB CHARSET=utf8;

DROP TABLE IF EXISTS teacher;
CREATE TABLE teacher(
tid VARCHAR(20) PRIMARY KEY COMMENT '教师ID',
tname VARCHAR(20) COMMENT '教师姓名'
)ENGINE=INNODB CHARSET=utf8;

DROP TABLE IF EXISTS course;
CREATE TABLE course(
cid VARCHAR(20) PRIMARY KEY COMMENT '课程ID',
cname VARCHAR(50)  COMMENT '课程名',
tid VARCHAR(20) COMMENT '任教教师ID'
)ENGINE=INNODB CHARSET=utf8;

DROP TABLE IF EXISTS SC;
CREATE TABLE sc(
sid VARCHAR(20) COMMENT '学生ID',
cid VARCHAR(20) COMMENT '课程ID',
score INT COMMENT '分数'
)ENGINE=INNODB CHARSET=utf8;

三、实战问题

1.查询“C01”课程比“C02”课程成绩高的所有学生的学号

-- 先分别查出学习"C01"、"C02"课程的学生ID和成绩
SELECT sid,score FROM sc WHERE cid='C01';
SELECT sid,score FROM sc WHERE cid='C02';
-- 再查询出"C01"大于"C02"课程的学生成绩
SELECT t1.sid FROM 
(SELECT sid,score FROM sc WHERE cid='C01') AS t1
INNER JOIN
(SELECT sid,score FROM sc WHERE cid='C02') AS t2
ON t1.sid=t2.sid WHERE t1.score>t2.score;
